class LinearRegression:
    
    def fit(self,X,y):
        '''
        
        
        Parameters
        X :  [,]
        y :  []
        '''
        # XX.copy()
        X = np.asmatrix(X.copy())
        # reshape() 
        y = np.asmatrix(y).reshape(-1,1) 
        
        # 
        self.w = (X.T * X).I * X.T * y

        
    def predict(self,X):
        '''
        
        
        returns : 
        '''
        
        X = np.asmatrix(X.copy())
        result = X * self.w
        return np.array(result).ravel()
